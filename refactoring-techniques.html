Change Reference to Value

No mapped smells
▶ Show example

Before:

public class Money {
    private double amount;
    private String currency;

    public void setAmount(double amount) {
        this.amount = amount;
    }

    public void setCurrency(String currency) {
        this.currency = currency;
    }
}
// usage: money.setAmount(50); — mutates in place

After:

public class Money {
    private final double amount;
    private final String currency;

    public Money(double amount, String currency) {
        this.amount = amount;
        this.currency = currency;
    }

    public Money withAmount(double amount) {
        return new Money(amount, this.currency);
    }

    public boolean equals(Object o) {
        if (!(o instanceof Money)) return false;
        Money m = (Money) o;
        return amount == m.amount && currency.equals(m.currency);
    }
}
// usage: money = new Money(50, "USD"); — immutable value

Change Value to Reference

No mapped smells
▶ Show example

Before:

// Each Order creates its own Customer copy
public class Order {
    private Customer customer;

    public Order(String customerName) {
        this.customer = new Customer(customerName);
    }
}
// Two orders for "Alice" have different Customer objects

After:

// Orders share a single Customer reference
public class Order {
    private Customer customer;

    public Order(String customerName) {
        this.customer = CustomerRegistry.get(customerName);
    }
}

public class CustomerRegistry {
    private static Map<String, Customer> customers = new HashMap<>();

    public static Customer get(String name) {
        return customers.computeIfAbsent(name, Customer::new);
    }
}
// Two orders for "Alice" share the same Customer

// Each caller recalculates derived values

double total = base + tax;

// ... in another method, same logic repeated

    }

Compose Method

No mapped smells
▶ Show example

Before:

public void addEmployee(List<Employee> employees, String name,
                        String role, double salary) {
    boolean valid = true;
    if (name == null || name.trim().isEmpty()) valid = false;
    if (salary < 0) valid = false;
    if (role == null) valid = false;
    if (!valid) throw new IllegalArgumentException();
    Employee emp = new Employee();
    emp.setName(name.trim());
    emp.setRole(role);
    emp.setSalary(salary);
    emp.setStartDate(LocalDate.now());
    emp.setId(generateId());
    employees.add(emp);
    auditLog.record("Added " + name);
    notifyHR(emp);
}

After:

public void addEmployee(List<Employee> employees, String name,
                        String role, double salary) {
    validate(name, role, salary);
    Employee emp = createEmployee(name, role, salary);
    employees.add(emp);
    notifyNewHire(emp);
}

private void validate(String name, String role, double salary) {
    if (name == null || name.trim().isEmpty())
        throw new IllegalArgumentException("Invalid name");
    if (role == null) throw new IllegalArgumentException("Invalid role");
    if (salary < 0) throw new IllegalArgumentException("Invalid salary");
}

private Employee createEmployee(String name, String role, double salary) {
    Employee emp = new Employee();
    emp.setName(name.trim());
    emp.setRole(role);
    emp.setSalary(salary);
    emp.setStartDate(LocalDate.now());
    emp.setId(generateId());
    return emp;
}

private void notifyNewHire(Employee emp) {
    auditLog.record("Added " + emp.getName());
    notifyHR(emp);
}

public double disabilityAmount(Employee emp) {
    if (emp.getSeniority() < 2) return 0;
    if (emp.getMonthsDisabled() > 12) return 0;
    if (emp.isPartTime()) return 0;
    // compute disability amount
    return emp.getSalary() * 0.6;
}

After:

public double disabilityAmount(Employee emp) {
    if (isNotEligibleForDisability(emp)) return 0;
    return emp.getSalary() * 0.6;

    return emp.getSeniority() < 2
        || emp.getMonthsDisabled() > 12
        || emp.isPartTime();
}

▶ Show example

Before:

// Subclasses duplicate composite-handling logic
public class SingleArticle extends Article {
    public double getPrice() { return price; }
}

    private List<Article> articles;

        double total = 0;

public abstract class CompositeArticle extends Article {

        return total;
    }

public class Bundle extends CompositeArticle {
    public double getPrice() { return getBasePrice(); }
}

public class DiscountBundle extends CompositeArticle {
    public double getPrice() { return getBasePrice() * 0.9; }
}

Extract Variable

No mapped smells
▶ Show example

Before:

public double price(Order order) {
    return order.getQuantity() * order.getItemPrice()
        - Math.max(0, order.getQuantity() - 500) * order.getItemPrice() * 0.05
        + Math.min(order.getQuantity() * order.getItemPrice() * 0.1, 100);
}

public double price(Order order) {
    double basePrice = order.getQuantity() * order.getItemPrice();
    double quantityDiscount = Math.max(0, order.getQuantity() - 500)

    return basePrice - quantityDiscount + shipping;
}

No mapped smells
▶ Show example

Before:

// Superclass hard-codes the type it creates
public class HTMLParser extends Parser {
    public void parse(String text) {

Move Accumulation to Collecting Parameter

No mapped smells
▶ Show example

Before:

// Building result by concatenating return values
public String renderTree(TreeNode node) {
    String result = "<" + node.getName() + ">";
    for (TreeNode child : node.getChildren()) {
        result += renderTree(child);
    }
    result += "</" + node.getName() + ">";
    return result; // O(n^2) string concatenation
}

After:

// Pass a collecting parameter to accumulate the result
public String renderTree(TreeNode node) {
    StringBuilder sb = new StringBuilder();
    renderTree(node, sb);
    return sb.toString();
}

private void renderTree(TreeNode node, StringBuilder sb) {
    sb.append("<").append(node.getName()).append(">");
    for (TreeNode child : node.getChildren()) {
        renderTree(child, sb);
    }
    sb.append("</").append(node.getName()).append(">");
}

No mapped smells
▶ Show example

public double priceOrder(Product product, int quantity,
                         ShippingMethod shipping) {
    double basePrice = product.getBasePrice() * quantity;
    double discount = Math.max(quantity - 500, 0) * product.getBasePrice() * 0.05;
    double shippingCost = quantity * shipping.getPerItemRate();

        shippingCost *= shipping.getDiscountThreshold();

    return basePrice - discount + shippingCost;
}

After:

public double priceOrder(Product product, int quantity,
                         ShippingMethod shipping) {
    PriceData priceData = calculatePriceData(product, quantity);
    return applyShipping(priceData, shipping);
}

private PriceData calculatePriceData(Product product, int quantity) {
    double basePrice = product.getBasePrice() * quantity;
    double discount = Math.max(quantity - 500, 0)
                      * product.getBasePrice() * 0.05;
    return new PriceData(basePrice, discount, quantity);
}

private double applyShipping(PriceData data, ShippingMethod shipping) {
    double shippingCost = data.quantity * shipping.getPerItemRate();
    if (data.basePrice - data.discount > 1000)
        shippingCost *= shipping.getDiscountThreshold();
    return data.basePrice - data.discount + shippingCost;
}
